local function scanRemotes()
    if _scanDone then return end
    _scanDone=true
    local rs=game:GetService("ReplicatedStorage")
    local hdNames={
        "HDAdminRemote","MainRemote","HDAdmin",
        "HDAdminEvent","AdminRemote","HD_Admin",
        "HDAdminRE","Commands"
    }
    local f3xNames={
        "F3XRemote","BuildRemote","Remote","BuildService"
    }
    for _,n in ipairs(hdNames)do
        local obj=rs:FindFirstChild(n)
        if obj and obj:IsA("RemoteEvent")then
            _hdRemote=obj break
        end
    end
    for _,n in ipairs(f3xNames)do
        local obj=rs:FindFirstChild(n)
        if obj and obj:IsA("RemoteEvent")then
            _f3xRemote=obj break
        end
    end
    local notif=Instance.new("Frame",sg)
    notif.Size=UDim2.new(0,300,0,36)
    notif.Position=UDim2.new(0.5,-150,1,10)
    notif.BackgroundColor3=BG
    notif.BorderSizePixel=2
    notif.BorderColor3=BORDER
    notif.ZIndex=999
    local ntxt=Instance.new("TextLabel",notif)
    ntxt.Size=UDim2.new(1,-8,1,0)
    ntxt.Position=UDim2.new(0,4,0,0)
    ntxt.BackgroundTransparency=1
    ntxt.TextColor3=TXT
    ntxt.Font=Enum.Font.GothamBold
    ntxt.TextSize=10
    ntxt.TextWrapped=true
    ntxt.TextXAlignment=Enum.TextXAlignment.Left
    ntxt.ZIndex=999
    local hds=_hdRemote and("OK:"..(_hdRemote.Name))or"NAO ACHOU"
    local f3s=_f3xRemote and("OK:"..(_f3xRemote.Name))or"NAO ACHOU"
    ntxt.Text="[SCAN] HD:"..hds.." | F3X:"..f3s
    T:Create(notif,TweenInfo.new(0.4),{
        Position=UDim2.new(0.5,-150,1,-50)
    }):Play()
    task.wait(5)
    T:Create(notif,TweenInfo.new(0.4),{
        Position=UDim2.new(0.5,-150,1,10)
    }):Play()
    task.wait(0.5)
    notif:Destroy()
end
task.spawn(scanRemotes)
local function hdCmd(cmd)
    if not _hdRemote then scanRemotes() end
    if _hdRemote then pcall(function()_hdRemote:FireServer(cmd)end) end
end
local function f3xFire(a,b,c,d)
    if not _f3xRemote then scanRemotes() end
    if _f3xRemote then pcall(function()_f3xRemote:FireServer(a,b,c,d)end) end
end
local function f3xProp(obj,prop,val)
    f3xFire("ChangeProp",obj,prop,val)
    pcall(function()
        if prop=="BrickColor" then obj.BrickColor=val
        elseif prop=="Material" then obj.Material=val
        elseif prop=="Anchored" then obj.Anchored=val
        end
    end)
end
